import random, sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        S = rng.randint(10, 80)
        if S % 2 != 0:
            continue
        if S // 2 < 5:
            continue
        break
    problem = (
        f"Two positive real numbers x and y satisfy x + y = {S}. "
        f"Compute their arithmetic mean (x + y)/2."
    )
    return problem, str(sympy.Integer(S // 2))

import random, sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        pairs = [(rng.randint(1, 9), rng.randint(1, 9)) for _ in range(4)]

        best = max(products)
        if best < 10:
            continue
        break
    listing = ", ".join(f"({a}, {b})" for a, b in pairs)
    problem = (
        f"Among the following pairs of positive integers, find the "
        f"largest value of the product x*y: {listing}."
    )

CROSSOVER_CHILD_PROGRAM_EXAMPLE_1:
```python
"""Crossover hybrid generator.

Parent A (algebra.arithmetic_mean): for x + y = S, the mean
(x + y)/2 = S/2 is fixed.
Parent B (algebra.product_extremum_discrete): pick the (x, y)
that maximises x*y.

Shared structure: the same pair (x, y) is both constrained by
A's mean condition and evaluated by B's product. The AM-GM
inequality
    (x + y)/2  >=  sqrt(x*y)
is the identity fusing them: A's mean bounds B's product, tight
at x = y, giving max x*y = (S/2)^2.

Why the solver is stuck without AM-GM: the feasible set is now
continuous, so B's "max over a list" move does not apply, and no
shorter handle exists.

Guards reject S odd (forces integer answer), (S/2)^2 <= 25 (too
small to require work), and S < 8 (optimum too close to the
boundary the solver might guess). Problem text states only S; it
never mentions AM-GM, the x = y optimum, or completing the square.
"""

import random, sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        S = rng.randint(8, 60)
        if S % 2 != 0:
            continue
        best = (S // 2) ** 2
        if best <= 25:
            continue

        scan = max(x * (S - x) for x in range(1, S))
        assert scan == best
        break

    problem = (
        f"Let x and y be positive real numbers with x + y = {S}. "
        f"Find the maximum possible value of the product x*y."
    )
    return problem, str(sympy.Integer(best))

CONCEPT_GROUP = "algebra"
CONCEPT_TYPE = "algebra.am_gm_product_max"
CONCEPT_REASON = "AM-GM fuses arithmetic_mean (constraint) with product_extremum_discrete (objective): the mean bounds the product, tight at x = y."
```

import math
import sympy
import random

def generate(seed):
    rng = random.Random(seed)
    while True:
        n = rng.randint(8, 16)
        k = rng.randint(3, n - 3)
        if k == n - k:
            continue
        answer = math.comb(n, k)
        if answer < 30:
            continue
        break
    problem = (
        f"Compute the binomial coefficient C({n}, {k}), i.e. the "
        f"number of ways to choose {k} items from {n} distinct items."
    )
    return problem, str(sympy.Integer(answer))

import sympy
import random

def generate(seed):
    rng = random.Random(seed)
    while True:
        r = rng.choice([2, 3, 4, 5])
        n = rng.randint(4, 8)
        if r == 2 and n < 5:
            continue
        total = (r ** n - 1) // (r - 1)
        if total < 50:
            continue
        break
    problem = (
        f"Compute the sum of the geometric series "
        f"1 + {r} + {r}^2 + ... + {r}^{n - 1}."
    )
    return problem, str(sympy.Integer(total))

CROSSOVER_CHILD_PROGRAM_EXAMPLE_2:
```python
"""Crossover hybrid generator.

Parent A (combinatorics.binomial_coefficient): C(n, k) counts
k-subsets of an n-set.
Parent B (sequence.geometric_sum): 1 + r + r^2 + ... + r^(n-1)
sums a geometric progression.

Shared structure: the single expansion (1 + r)^n is at once
A's generating object — its coefficients are the C(n, k) — and
B's value when those coefficients are weighted by powers of r.
The binomial theorem
    (1 + r)^n  =  sum_{k=0..n} C(n, k) * r^k
fuses them on the same polynomial: the sum the solver wants is
a weighted geometric-style sum whose weights are exactly A's
binomials. Substituting x = r collapses the whole sum to a
single power (1 + r)^n.

Why the solver is stuck without the binomial theorem: A alone
gives one coefficient at a time, requiring n separate binomial
evaluations and a final weighted sum; B alone applies only to
unit weights. Recognising the sum as (1 + r)^n is the only
short path.

Guards reject r = 1 (collapses to A's row sum 2^n), small n
where the term-by-term sum is feasible, and answers that are
named powers the solver might recognise by inspection.
Problem text lists the series explicitly; it never mentions
the binomial theorem or that the answer is a single power.
"""

import sympy
import random

def generate(seed):
    rng = random.Random(seed)
    while True:
        r = rng.choice([2, 3, 4, 5])
        n = rng.randint(6, 11)
        if r == 1:
            continue
        if n <= 4:
            continue

        answer = (1 + r) ** n
        if answer in (64, 128, 256, 512, 1024):
            continue

        check = sum(sympy.binomial(n, k) * r ** k for k in range(n + 1))
        assert int(check) == answer
        break

    terms = " + ".join(
        f"C({n},{k})*{r}^{k}" if k > 0 else f"C({n},0)"
        for k in range(n + 1)
    )
    problem = (
        f"Compute the value of the sum {terms}, where C(n, k) "
        f"denotes the binomial coefficient. State the answer as "
        f"a single integer."
    )
    return problem, str(sympy.Integer(answer))

CONCEPT_GROUP = "combinatorics"
CONCEPT_TYPE = "combinatorics.binomial_theorem_sum"
CONCEPT_REASON = "The binomial theorem fuses binomial_coefficient (the weights) with geometric_sum (the power structure): the weighted sum collapses to the single power (1 + r)^n."
```
