PARENT_PROGRAM_EXAMPLE_1:
```python
import random
import sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        a1 = rng.randint(2, 20)
        d = rng.randint(2, 12)
        n = rng.randint(8, 25)
        answer = a1 + (n - 1) * d
        if answer < 50:
            continue
        break
    problem = (
        f"An arithmetic sequence has first term {a1} and common "
        f"difference {d}. Compute its {n}-th term."
    )
    return problem, str(sympy.Integer(answer))

CONCEPT_GROUP = "sequence"
CONCEPT_TYPE = "sequence.arithmetic_nth_term"
```

MUTATED_PROGRAM_EXAMPLE_1:
```python
"""In-breadth mutation generator.

Parent (sequence.arithmetic_nth_term): a_n = a_1 + (n-1)*d, a
single linear evaluation in a discrete progression.
Child (geometry.chord_length_from_center): chord length given
radius r and centre-distance d, answered as an irrational surd.

The jump is in-breadth: parent and child share no domain, no
move, no vocabulary. The solver must drop the radius to a chord
endpoint, see that radius, centre-perpendicular, and half-chord
form a right triangle, apply Pythagoras to get half-chord =
sqrt(r^2 - d^2), and double. Coordinate geometry is the only
alternative and runs several times longer.

Guards reject r^2 - d^2 a perfect square (integer chord,
guessable), radicand m > 30 (artificial), and gcd(r, d) > 1
(rescales to a smaller named case). Problem text states only r
and d; Pythagoras and the bisection are never mentioned.
"""

import random
import sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        r = rng.randint(5, 15)
        d = rng.randint(1, r - 1)

        diff = r * r - d * d
        m, k = diff, 1
        for q in sympy.primerange(2, 20):
            while m % (q * q) == 0:
                m //= q * q
                k *= q

        if m == 1:
            continue
        if m > 30:
            continue
        if sympy.gcd(r, d) > 1:
            continue

        coef = 2 * k
        answer = f"{coef}*sqrt({m})"
        break

    problem = (
        f"A circle has radius {r}. A chord of the circle lies at "
        f"perpendicular distance {d} from the centre. Compute "
        f"the length of the chord in exact form."
    )
    return problem, answer

CONCEPT_GROUP = "geometry"
CONCEPT_TYPE = "geometry.chord_length_from_center"
CONCEPT_REASON = "Mutation jumps from sequence to geometry: discrete progression yields to continuous Pythagorean decomposition, with no shared move between parent and child."
```

PARENT_PROGRAM_EXAMPLE_2:
```python
import random
import sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        a = rng.randint(3, 15)
        b = rng.randint(3, 15)
        if a == b:
            continue
        area = a * b
        if area < 20:
            continue
        break
    problem = (
        f"A rectangle has side lengths {a} and {b}. "
        f"Compute its area."
    )
    return problem, str(sympy.Integer(area))

CONCEPT_GROUP = "geometry"
CONCEPT_TYPE = "geometry.rectangle_area"
```

MUTATED_PROGRAM_EXAMPLE_2:
```python
"""In-breadth mutation generator.

Parent (geometry.rectangle_area): one multiplication of two
side lengths to get a continuous area.
Child (number_theory.divisor_count): given an integer N,
report the number of positive divisors d(N).

The jump is in-breadth: parent and child share no domain, no
move, no vocabulary. The solver must factor N into primes
N = p1^e1 * p2^e2 * ... * pk^ek, recall the multiplicative
formula d(N) = (e1+1)(e2+1)...(ek+1), and evaluate it. Trial
division up to sqrt(N) is the only alternative and grows with
N, while the formula collapses the count to a small product of
exponents-plus-one.

Guards reject N prime or prime-power (formula degenerates to a
single factor, no real chain), N with fewer than three distinct
prime factors (one multiplication is not a chain), and answers
in {1, 2, 3, 4} (named-trivial counts the solver guesses by
inspection). Problem text states only N; factorisation and the
exponent formula are never mentioned.
"""

import random
import sympy

def generate(seed):
    rng = random.Random(seed)
    while True:
        N = rng.randint(200, 5000)
        factors = sympy.factorint(N)

        if len(factors) < 3:
            continue

        count = 1
        for e in factors.values():
            count *= (e + 1)

        if count <= 4:
            continue
        if count > 36:
            continue

        break

    problem = (
        f"Let N = {N}. Compute the number of positive integers "
        f"that divide N (including 1 and N itself)."
    )
    return problem, str(sympy.Integer(count))

CONCEPT_GROUP = "number_theory"
CONCEPT_TYPE = "number_theory.divisor_count"
CONCEPT_REASON = "Mutation jumps from geometry to number_theory: continuous side-length multiplication yields to prime-exponent enumeration via d(N) = prod(e_i + 1), with no shared move between parent and child."
```
